Fill in the empty values with the means for the given interval Do so by combining the dataframe with the empty steps with the dataframe with the means, and keep only the calculated steps in a new dataframe.

join_string <- "select
              meansteps.steps
              , emptysteps.date
              , emptysteps.interval
              from emptysteps
                left join meansteps
                on emptysteps.interval = meansteps.interval"

joinedsteps <- sqldf(join_string, stringsAsFactors = FALSE)
# summary(joinedsteps)

newactivity <- rbind(filledsteps, joinedsteps)
# summary(newactivity)
